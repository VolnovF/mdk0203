## Физические константы и специальные функции


```python
def get_wv(D0):
    h = 6.626e-34
    c = 3.0e8
    N_A = 6.022e23
    
    E_molecule = (D0 * 1e3) / N_A
    lambda_wave = h * c / E_molecule
    lambda_wave_nm = lambda_wave * 1e9
    return lambda_wave_nm

print(get_wv(500))
```
